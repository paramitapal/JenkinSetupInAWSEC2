sudo yum update -y
sudo yum install -y docker
sudo service docker start
sudo systemctl enable docker
sudo curl -L "https://github.com/docker/compose/releases/download/1.23.1/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose
docker pull jenkins/jenkins
docker info | grep -i root
mkdir jenkins-data
cd jenkins-data
vi docker-compose.yml



(Add the following in the yml file:)
version: '3'
services:
  jenkins:
    container_name: jenkins
    image: jenkins/jenkins
    ports:
      - "8080:8080"
    volumes:
      - $PWD/jenkins_home:/var/jenkins_home
    networks:
      - net
networks:
  net:


Notes: 
1. create a directory "jenkins_home" at the same level where the yml file is.
2. sudo chown 1000:1000 jenkins_home(make sure the jenkins id is 1000, to check the jenkins id, write : "id" in your bash)
3. Up the server: docker-compose up -d
4. Check if the docker is up: docker ps
5. See the log: docker logs -f jenkins(copy the initial password from the log, it can also be found in "jenkins_home/secrets/")
